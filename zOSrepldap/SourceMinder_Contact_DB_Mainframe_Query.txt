SELECT P.PROD_NAME,P.RELEASE,P.PROD_STAT,PM.SRC_MNGMT_TOOL,PM.ENDEVOR_PRODUCT,PMC.MAINFRAME_SRC_LOC_DSNAME,
       P.APPROVERS_PMFKEY,E1.PMFKEY AS ERC,E2.PMFKEY AS SEM,P.SCAS
FROM MASTER.PRODUCT_MASTER PM
JOIN MASTER.PRODUCT P ON (PM.PRODUCT_ID=P.PRODUCT_ID)
JOIN MASTER.PRODUCT_MASTER_CHANGES PMC ON (PMC.ATTESTATION_ID=PM.ATTESTATION_ID)
JOIN MASTER.EMPLOYEE E1 ON (E1.EMPLOYEE_ID=P.ERC_ID)
JOIN MASTER.EMPLOYEE E2 ON (E2.EMPLOYEE_ID=P.SEM_ID)
WHERE PM.ENDEVOR_PRODUCT is null
  AND PMC.MAINFRAME_SRC_LOC_DSNAME not in ('N/A','')
  AND PM.SRC_MNGMT_TOOL not in ('N/A','Harvest','')
